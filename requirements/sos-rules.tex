
\infrule{Input}{
  \eta' = \eta\setto{\vec{x}}{\vec{v}}
  \andalso
  \vec{x} = \Varin
  }{
  (S,\eta,\inputstatus) \transin{v}
  (S,\eta',\calcstatus{\act(S)})
  }

\\

\infrule{Act}{%
  a \in A
  \andalso
  \eta' = \sem{}{a}(\eta)
  }{
  (S,\eta,\calcstatus{A}) 
  \trans{}
  (S,\eta',\calcstatus{A\without a})
  }

\infrule{No-Act}{
  \andalso
  T = \{
  S_s \edge{g} S_t \in T
  \suchthat
  S_s \subseteq S
  \land
  \sem{\eta}{g} = \ltrue \}
  }{
  (S,\eta,\calcstatus{\emptyset})
  \trans{}
  (S,\eta,\outputstatus{T})
  }

\\

\infrule{Trans}{%
  t =  S_s \edge{g} S_t
 \in T
  \andalso
  S_s  \subseteq S
  \andalso
  S' = (S \union S_t) \without S_s
  }{
  (S,\eta,\outputstatus{T})
  \trans{}
  (S',\eta,\outputstatus{T \without t})
  }

\\
\infrule{No-Trans}{%
  S_s \edge{g} S_t
 \in T
  \andalso
  S_s  \not\subseteq S
  \andalso
  }{
  (S,\eta,\outputstatus{T})
  \trans{}
  (S,\eta,\outputstatus{T \without t})
  }

\\

\infrule{Output}{%
  \vec{v} = \sem{\eta}{\vec{x}}
  \andalso
  \vec{x} = \Varout
  }{%
  (S,\eta, \outputstatus{\emptyset}) \transout{\vec{v}}
  (S,\eta,\inputstatus)
  }
